#include <stdio.h>
void ratioCalc(int data[], int len) 
{
    int posCount = 0, negCount = 0, zeroCount = 0;
    for (int i = 0; i < len; i++) 
    {
        if (data[i] > 0)
            posCount++;
        else if (data[i] < 0)
            negCount++;
        else
            zeroCount++;
    }
    printf("%.6f\n", (double)posCount / len);
    printf("%.6f\n", (double)negCount / len);
    printf("%.6f\n", (double)zeroCount / len);
}
int main() 
{
    int size;
    scanf("%d", &size);
    int arr[size];
    for (int i = 0; i < size; i++) 
    {
        scanf("%d", &arr[i]);
    }
    ratioCalc(arr, size);
    return 0;
}
