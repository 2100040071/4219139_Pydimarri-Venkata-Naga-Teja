#include <stdio.h>
#include <stdbool.h>
#include <string.h>
#include <ctype.h>
char* checkPangram(char* text) 
{
    bool lettersFound[26] = {false};
    int pos;
    for (int i = 0; text[i] != '\0'; i++) 
    {
        if (isalpha(text[i])) 
        {
            pos = tolower(text[i]) - 'a';
            lettersFound[pos] = true;
        }
    }
    for (int k = 0; k < 26; k++) 
    {
        if (!lettersFound[k]) 
        {
            return "not pangram";
        }
    }
    return "pangram";
}
int main() 
{
    char sentence[1001];
    fgets(sentence, sizeof(sentence), stdin);
    printf("%s\n", checkPangram(sentence));
    return 0;
}
