#include <stdio.h>
#include <string.h>
char* xorStrings(char *str1, char *str2) {
    static char output[10005];
    int len = strlen(str1);
    for (int i = 0; i < len; i++) {
        if (str1[i] == str2[i])
            output[i] = '0';
        else
            output[i] = '1';
    }
    output[len] = '\0';
    return output;
}
int main() {
    char first[10005], second[10005];
    scanf("%s", first);
    scanf("%s", second);
    printf("%s\n", xorStrings(first, second));
    return 0;
}
