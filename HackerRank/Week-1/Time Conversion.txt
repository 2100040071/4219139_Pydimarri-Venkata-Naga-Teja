#include <stdio.h>
#include <string.h>
void convertTime(char t[]) 
{
    int hour, minute, second;
    char period[3];
    sscanf(t, "%d:%d:%d%2s", &hour, &minute, &second, period);
    if (strcmp(period, "AM") == 0) 
    {
        if (hour == 12) hour = 0; 
    } else if (strcmp(period, "PM") == 0) 
    {
        if (hour != 12) hour += 12; 
    }
    printf("%02d:%02d:%02d\n", hour, minute, second);
}
int main() 
{
    char input[12];
    scanf("%s", input);
    convertTime(input);
    return 0;
}
