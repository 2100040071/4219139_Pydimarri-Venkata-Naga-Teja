#include <stdio.h>
#include <stdlib.h>
int* generateFrequency(int numElements, int* numbers, int* freqSize) 
{
    static int freqTable[100] = {0}; 
    for (int idx = 0; idx < numElements; idx++) 
    {
        int val = numbers[idx];
        if (val >= 0 && val < 100) 
        {
            freqTable[val]++;
        }
    }
    *freqSize = 100; 
    return freqTable;
}
int main() 
{
    int totalNums;
    scanf("%d", &totalNums);
    int* values = malloc(sizeof(int) * totalNums);
    for (int i = 0; i < totalNums; i++) 
    {
        scanf("%d", &values[i]);
    }
    int tableSize;
    int* frequency = generateFrequency(totalNums, values, &tableSize);
    for (int i = 0; i < tableSize; i++) 
    {
        printf("%d ", frequency[i]);
    }
    printf("\n");
    free(values);
    return 0;
}
