#include <stdio.h>
#include <stdlib.h>
int findDiagonalDiff(int matrix[][100], int n) 
{
    int leftDiagSum = 0, rightDiagSum = 0;
    for (int row = 0; row < n; row++) 
    {
        leftDiagSum += matrix[row][row]; 
        rightDiagSum += matrix[row][n - 1 - row]; 
    }
    return abs(leftDiagSum - rightDiagSum); 
}
int main() 
{
    int size;
    scanf("%d", &size);
    int mat[100][100];
    for (int r = 0; r < size; r++) 
    {
        for (int c = 0; c < size; c++) 
        {
            scanf("%d", &mat[r][c]);
        }
    }
    int difference = findDiagonalDiff(mat, size);
    printf("%d\n", difference);
    return 0;
}
