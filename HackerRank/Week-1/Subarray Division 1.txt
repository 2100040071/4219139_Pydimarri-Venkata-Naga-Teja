#include <stdio.h>
int countMatchingSegments(int arr[], int arrSize, int desiredSum, int segmentLen) 
{
    int matchCount = 0;
    for (int start = 0; start <= arrSize - segmentLen; start++) 
    {
        int sum = 0;
        for (int j = 0; j < segmentLen; j++) 
        {
            sum += arr[start + j];
        }
        if (sum == desiredSum) 
        {
            matchCount++;
        }
    }
    return matchCount;
}
int main() 
{
    int n;
    scanf("%d", &n);
    int values[n];
    for (int i = 0; i < n; i++) 
    {
        scanf("%d", &values[i]);
    }
    int targetSum, targetLen;
    scanf("%d %d", &targetSum, &targetLen);
    int answer = countMatchingSegments(values, n, targetSum, targetLen);
    printf("%d\n", answer);
    return 0;
}
