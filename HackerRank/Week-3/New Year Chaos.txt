#include <stdio.h>
void countBribes(int line[], int size) 
{
    int moves = 0;
    for (int i = 0; i < size; i++) 
    {
        if (line[i] - (i + 1) > 2) 
        {
            printf("Too chaotic\n");
            return;
        }
        int start = (line[i] - 2 > 0) ? line[i] - 2 : 0;
        for (int j = start; j < i; j++) 
        {
            if (line[j] > line[i]) 
            {
                moves++;
            }
        }
    }
    printf("%d\n", moves);
}
int main() 
{
    int testCases;
    scanf("%d", &testCases);
    while (testCases--) 
    {
        int size;
        scanf("%d", &size);
        int arr[size];
        for (int i = 0; i < size; i++) 
        {
            scanf("%d", &arr[i]);
        }
        countBribes(arr, size);
    }
    return 0;
}
