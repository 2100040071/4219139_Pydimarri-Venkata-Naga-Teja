#include <stdio.h>
#include <string.h>
const char* checkValidity(char* str) 
{
    int count[26] = {0};
    int length = strlen(str);
    for (int i = 0; i < length; i++) 
    {
        count[str[i] - 'a']++;
    }
    int low = 0, high = 0;
    int lowFreqCount = 0, highFreqCount = 0;
    for (int i = 0; i < 26; i++) 
    {
        if (count[i] > 0) 
        {
            if (low == 0 || count[i] < low) low = count[i];
            if (count[i] > high) high = count[i];
        }
    }
    for (int i = 0; i < 26; i++) 
    {
        if (count[i] == low) lowFreqCount++;
        else if (count[i] == high) highFreqCount++;
    }
    if (low == high) return "YES";  
    if (low == 1 && lowFreqCount == 1 && (highFreqCount * high + 1 == length)) return "YES";  
    if ((high - low == 1) && highFreqCount == 1) return "YES";
    return "NO";
}
int main() 
{
    char input[100005];
    scanf("%s", input);
    printf("%s\n", checkValidity(input));
    return 0;
}
