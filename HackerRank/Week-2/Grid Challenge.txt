#include <stdio.h>
#include <stdlib.h>
#include <string.h>
int charCompare(const void *a, const void *b) 
{
    return (*(char *)a - *(char *)b);
}
void sortStringRow(char *str) 
{
    qsort(str, strlen(str), sizeof(char), charCompare);
}
int checkColumnsSorted(char **matrix, int rows, int cols) 
{
    for (int c = 0; c < cols; c++) 
    {
        for (int r = 1; r < rows; r++) 
        {
            if (matrix[r][c] < matrix[r - 1][c]) 
            {
                return 0;
            }
        }
    }
    return 1;
}
char* validateGrid(char **matrix, int rows) 
{
    for (int r = 0; r < rows; r++) 
    {
        sortStringRow(matrix[r]);
    }
    if (checkColumnsSorted(matrix, rows, strlen(matrix[0]))) 
    {
        return "YES";
    }
    return "NO";
}
int main() 
{
    int testCases;
    scanf("%d", &testCases);
    while (testCases--) 
    {
        int size;
        scanf("%d", &size);
        char **matrix = (char **)malloc(size * sizeof(char *));
        for (int i = 0; i < size; i++) 
        {
            matrix[i] = (char *)malloc((size + 1) * sizeof(char));
            scanf("%s", matrix[i]);
        }
        char *res = validateGrid(matrix, size);
        printf("%s\n", res);
        for (int i = 0; i < size; i++) 
        {
            free(matrix[i]);
        }
        free(matrix);
    }
    return 0;
}
