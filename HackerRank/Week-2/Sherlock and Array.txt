#include <stdio.h>
char* checkBalancedArray(int nums[], int length) 
{
    int total = 0, left = 0;
    for (int i = 0; i < length; i++) 
    {
        total += nums[i];
    }
    for (int i = 0; i < length; i++) 
    {
        total -= nums[i];  
        if (left == total)
        {
            return "YES";
        }
        left += nums[i];
    }
    return "NO";
}
int main() 
{
    int cases;
    scanf("%d", &cases);
    while (cases--) 
    {
        int size;
        scanf("%d", &size);
        int arr[size];
        for (int i = 0; i < size; i++) 
        {
            scanf("%d", &arr[i]);
        }
        char* ans = checkBalancedArray(arr, size);
        printf("%s\n", ans);
    }
    return 0;
}
